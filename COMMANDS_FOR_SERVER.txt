===============================================
КОМАНДЫ ДЛЯ ВЫПОЛНЕНИЯ НА СЕРВЕРЕ FIRST VDS
===============================================

ВАЖНО: Выполняйте команды последовательно после подключения к серверу через SSH

===============================================
ШАГ 1: ПРОВЕРКА СИСТЕМЫ (ОПЦИОНАЛЬНО)
===============================================

# Скачать и запустить скрипт проверки
cd /root
wget https://raw.githubusercontent.com/uShure/telegram-ai-sales-agent/main/check-before-cleanup.sh
chmod +x check-before-cleanup.sh
./check-before-cleanup.sh

===============================================
ШАГ 2: ПОЛНАЯ ОЧИСТКА И УСТАНОВКА
===============================================

# Скачать и запустить основной скрипт установки
cd /root
wget https://raw.githubusercontent.com/uShure/telegram-ai-sales-agent/main/cleanup-and-deploy.sh
chmod +x cleanup-and-deploy.sh
sudo ./cleanup-and-deploy.sh

# ВНИМАНИЕ: Скрипт попросит подтверждение!
# Введите "yes" только если готовы удалить ВСЕ данные на сервере

===============================================
ШАГ 3: НАСТРОЙКА API КЛЮЧЕЙ
===============================================

# Вариант А: Интерактивная настройка (рекомендуется)
cd /root
wget https://raw.githubusercontent.com/uShure/telegram-ai-sales-agent/main/quick-setup.sh
chmod +x quick-setup.sh
./quick-setup.sh

# Вариант Б: Ручная настройка
bot config

# Заполните:
# API_ID=ваш_api_id (число)
# API_HASH=ваш_api_hash (32 символа)
# PHONE_NUMBER=+79991234567 (ваш номер)
# OPENAI_API_KEY=sk-... (ключ OpenAI)
# TELEGRAM_BOT_TOKEN=123456789:ABC... (токен бота)

===============================================
ШАГ 4: ЗАПУСК БОТА
===============================================

# Запустить бота
bot start

# Проверить статус
bot status

# Посмотреть логи
bot logs

===============================================
ПОЛЕЗНЫЕ КОМАНДЫ ПОСЛЕ УСТАНОВКИ
===============================================

bot start     # Запустить бота
bot stop      # Остановить бота
bot restart   # Перезапустить бота
bot status    # Проверить статус
bot logs      # Просмотр логов в реальном времени
bot errors    # Просмотр ошибок
bot monitor   # Мониторинг системы
bot backup    # Создать резервную копию
bot update    # Обновить бота из GitHub
bot config    # Редактировать конфигурацию

===============================================
ЕСЛИ ЧТО-ТО ПОШЛО НЕ ТАК
===============================================

# Проверить логи ошибок
bot errors
journalctl -u telegram-bot -n 50

# Проверить конфигурацию
cat /home/telegram-bot/app/.env

# Попробовать запустить вручную
cd /home/telegram-bot/app
sudo -u telegram-bot bun run src/index.ts

# Перезапустить бота
bot restart

===============================================
БЫСТРАЯ КОМАНДА ВСЕ-В-ОДНОМ (ОПАСНО!)
===============================================

# Эта команда выполнит ВСЮ установку автоматически
# ИСПОЛЬЗУЙТЕ ТОЛЬКО если уверены!

cd /root && \
wget https://raw.githubusercontent.com/uShure/telegram-ai-sales-agent/main/cleanup-and-deploy.sh && \
chmod +x cleanup-and-deploy.sh && \
echo "yes" | sudo ./cleanup-and-deploy.sh
